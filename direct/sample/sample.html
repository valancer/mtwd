Select box js 사용법

- 페이지 로딩시 공통스크립트에서 commonJs.setSelectBox("select");가 실행되어 모든 select박스를 생성합니다.

- 추가 생성 및 instance() 사용법 :

commonJs.setSelectBox("셀렉트박스 selector"); //생성
var ins=$("생성된 셀렉트박스 selector").getInstance(); //getInstance
ins.setIndexSB(인덱스번호); //선택(by Index)
ins.setValueSB(값); //선택(by Value)
ins.setTextSB(텍스트); //선택(by Text)
ins.getIndexSB(); //현재 인덱스 반환
ins.getValueSB(); //현재 값 반환
ins.getTextSB(); //현재 텍스트 반환
ins.getInfoSB(); //object형태로{index, value, text}를 반환한다.
ins.resetSB(); //셀렉트박스 재생성
ins.removeSB(); //생성된 셀렉트박스 디자인 삭제(생성되기전상태로 돌아가 기본select박스만 남는다.)
ins.disabledSB(value); //셀렉트 박스 disabled 설정(value값 true인경우 disabled 처리되고 false인경우 disabled을 해제한다.)

- 셀렉트박스 값 변경 이벤트:$("생성된셀렉트박스 selector").bind('change.selectbox', function(){실행 코드});
